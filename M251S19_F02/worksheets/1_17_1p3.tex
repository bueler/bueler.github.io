\begin{aproblems}
\aproblem Graph $y=x^3$ and $y=x^{\frac 1 3}$ on the same axes.  Add the line $y=x$ as a dashed line.  (\emph{This illustrates the idea that the graph of the inverse function is the reflection of the graph of the function.})
\vfill

\aproblem Graph $y=\sqrt{x}$, $y=\sqrt{x-1}$, and $y=\sqrt{-x}$; use three graphs side-by-side.
\vfill

\aproblem Graph $y=\sin(x)$ and $y=\cos(x)$ on the same graph over the interval $[-\pi,\pi]$. Label the points $-\pi$, $-\pi/2$, $0$, $\pi/2$, $\pi$ on the $x$-axis.
\vfill

\aproblem Graph $y=\cos(2x)$ and $y=2\cos(x)$ over the interval $[0,2\pi]$
on the same graph. 
Label the points $0$, $\pi/2$, $\pi$, $3\pi/2$ and $2\pi$ on the $x$-axis.
\vfill

\newpage
\aproblem Graph $y=\sin(|x|)$ and $y=|\sin(x)|$ over the interval $[-2\pi,2\pi]$
on adjacent graphs.
\vfill

\aproblem Graph $y=(0.1)^x$, $y=e^x$, $y=2 e^x$ on the same graph.
\vfill

\aproblem Graph $y=e^{-|x|}$.
\vfill

\aproblem Graph $y=\sin(x^2)$ and $y=\sin(1/x)$ on adjacent graphs. \hfill \emph{(challenging)} 
\vfill
\end{aproblems}
