\aproblem Sketch the graph $y=f(x)$ of a function which has all of the following properties; do not worry about any \emph{formula} for $f(x)$:

\hspace{-0.35in}
\begin{minipage}{0.6\textwidth}
\renewcommand{\labelenumi}{(\alph{enumi})}
\begin{enumerate}
\item $f(0)=3$
\item $\lim_{x\to 0} f(x) = 0$
\item $\lim_{x\to -2^-} f(x) = 1$
\item $\lim_{x\to -2^+} f(x) = -1$
\item $\lim_{x\to 5^-} f(x)$ d.n.e.
\item $\lim_{x\to 5^+} f(x) = -\infty$
\item the domain of $f$ is $(-\infty,\infty)$
\end{enumerate}
\end{minipage}
\vfill

\aproblem  Evaluate the limit, if it exists:
    $$\lim_{h\to 0} \frac{(2+h)^3 - 8}{h} = \hspace{4.0in}$$
\vfill

\aproblem  Evaluate the limit, if it exists:
    $$\lim_{u\to 2} \frac{\sqrt{4u+1} - 3}{u-2} = \hspace{4.0in}$$
\vfill

\newpage
\aproblem  Evaluate the limit, if it exists:
    $$\lim_{t\to 0} \left(\frac{1}{t\sqrt{1+t}} - \frac{1}{t}\right) = \hspace{4.0in}$$
\vfill

\aproblem  Evaluate the limit, if it exists:
    $$\lim_{x\to 3} \frac{\frac{1}{x} - \frac{1}{3}}{x-3} = \hspace{4.0in}$$
\vfill

\aproblem  Evaluate the limits, if they exist, and otherwise explain why they do not:

\hspace{-0.35in}
\begin{minipage}{0.6\textwidth}
\renewcommand{\labelenumi}{(\alph{enumi})}
\begin{enumerate}
\item
    $$\lim_{x\to 0^-} \left(\frac{1}{x} - \frac{1}{|x|}\right) = \hspace{4.0in}$$
\item
    $$\lim_{x\to 0^+} \left(\frac{1}{x} - \frac{1}{|x|}\right) = \hspace{4.0in}$$
\end{enumerate}
\end{minipage}\vfill

\aproblem  \emph{Challenge problem.}  Consider the following function:
    $$f(x) = \begin{cases} 1, & \text{ if } x \text{ is rational} \\
                           0, & \text{ if } x \text{ is irrational}\end{cases} \hspace{3.0in}$$
Evaluate the limit $\lim_{x\to 0} f(x)$ if it exists.  If it does not exist, explain why.
